import Decorator from '@/components/decorator-doc-entry.astro'

<Decorator name="@required" args="boolean">
  <Fragment slot="description">
    toggles whether the setting a value for the item is required.
    
    Both `undefined` or an `empty` string will be considered invalid.
  </Fragment>
  <Fragment slot="notes">
    default value can be toggled using the `@defaultRequired` root decorator
  </Fragment>
</Decorator>
<Decorator name="@optional" args="boolean">
  <Fragment slot="description">
    Opposite of `@required`. Equivalent to writing `@required=false`.
  </Fragment>
</Decorator>

<Decorator name="@sensitive" args="boolean">
  <Fragment slot="description">
    toggles whether the item should be considered sensitive (and must be protected from leaks)
  </Fragment>
  <Fragment slot="notes">
    default value can be toggled using the `@defaultSensitive` root decorator
  </Fragment>
</Decorator>

<Decorator name="@example" args="any">
  <Fragment slot="description">
    provide an example value for the item
  </Fragment>
</Decorator>

<Decorator name="@type" args="enum">
  <Fragment slot="description">
    sets the type of the item, will default to inferring the type from a static value if set, or `string` otherwise.

    <br/>
    See [data types reference](/reference/data-types) for more details.
  </Fragment>
  <Fragment slot="notes">
    some types take additional arguments, see below for more details
  </Fragment>
</Decorator>

<Decorator name="@docsUrl" args="string (url)">
  <Fragment slot="description">
    Where to find related documentation.
  </Fragment>
</Decorator>
