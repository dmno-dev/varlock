<Decorator name="@defaultRequired" args="boolean">
  <Fragment slot="description">
    Sets the the default state of each item being required. Can be overridden on individual items using `@required` or `@optional`.
  </Fragment>
</Decorator>

<Decorator name="@defaultSensitive" args="boolean">
  <Fragment slot="description">
    Sets the the default state of each item being treated sensitive. Can be overridden on individual items using `@sensitive`.
  </Fragment>
</Decorator>

<Decorator name="@generateTypes" args="lang=string, path=string" isFnCall>
  <Fragment slot="description">
    Enables automatic type generation based on your schema.
    <br/>
    - `lang`: Language to generate types for. Currently only `ts` (typescript) is supported
    - `path`: Relative filepath to output generated type file
  </Fragment>
</Decorator>
